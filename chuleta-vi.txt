COMANDOS NATIVOS EX
- :set wm=[n]		// Margen derecho n donde se inserta un salto de línea automático.
- :set nu \ set nonu	// Visualizar / quitar los números de línea.
- :set nowrapscan	// Evita continuar la búsqueda por el otro lado al llegar al final o inicio del archivo.
- :preserve / pre	// Preservar en un buffer de autoguardado el estado actual del archivo.
- :vi			// Volver al editor vi desde ex.
- :[n]+[m] \ [n]-[m]	// Posición relativa de m líneas hacia adelante \ atrás desde la línea n. Si no hay n la posiciónrelativa es a partir de la línea actual.
- :[n],[m]		// Líneas desde la posición n a la m. Si la m es posición relativa (+/-) se calcula a partir de la línea actual.
- :[n];[m]		// Líneas desde la posición n a la m. Si la m es posición relativa (+/-) se calcula a partir de n.
			// Si no hay m es por defecto la línea n.
- :[n]			// Mover el cursor a la línea n.
- :/patrón/		// Línea donde se encuentra la primera ocurrencia del patrón.
- :.			// Línea actual del archivo.
- :$			// Última línea del archivo.
- :%			// Todas las líneas del archivo.
- :args / ar		// Mostrar los nombres de archivos introducidos en la línea de comandos. Entre [] el archivo actual.
- :n 			// Mostrar el buffer del siguiente archivo de la línea de comandos. 
- :rewind / rew		// Mostrar el buffer del primer archivo de la línea de comandos.
- :e archivo		// Abrir un archivo cargando su contenido en un nuevo buffer.
- :e #			// Mostrar el buffer asignado con la etiqueta #. % equivale al nombre del archivo del buffer actual.
- :e!			// Descarta los cambios realizados en el buffer actual volviendo a la última versión guardada.
- :buffer [n] / b [n]	// Mostrar el buffer n
- :buffers / ls		// Mostrar un listado de todos los buffers.
- :bnext / bn		// Mostrar el siguiente buffer de la lista.
- :bprevious / bp	// Mostrar el buffer previo de la lista.
- :bfirst / bf		// Mostrar el primer buffer de la lista.
- :blast / bl		// Mostrar el último buffer de la lista.
- :bdelete / bd		// Eliminar el buffer.
// Antes de los siguientes comandos se puede indicar el número o números de líneas a aplicar.
- :d			// Borrar la línea actual.
- :m[l]			// Mover la línea actual a la posición siguiente a la línea l.
- :co[l] \ t[l]		// Copiar la línea actual a la posición siguiente a la línea l. 
- :# / p		// Mostrar temporalmente la línea actual.
- :s/p1/p2/		// Sustituir en la línea actual el primer patrón encontrado p1 por p2.
- :s/p1/p2/g		// Sustituir en la línea actual todo patrón p1 por p2.
- :=			// Mostrar el número total de líneas del archivo.
- :[n]=			// Mostrar el número de línea de la línea n. Si n es un cjto de líneas, se muestra el número de la última.
- :g/patrón		// Mover a la última ocurrencia del patrón en el archivo. Puede indicarse antes las líneas a buscar.
- :g/patrón/p		// Mostrar todas las líneas que contienen el patrón.
- :g/patrón/nu		// Mostrar todas las líneas que contienen el patrón junto con el número de línea.
- :g!/patrón		// Mostrar todas las líneas que no contienen el patrón. 
- :c1 | c2		// Combinar varios comandos.
- :w [archivo]		// Guardar el buffer en el archivo, haya sido modificado o no. Si no se indica archivo, se guarda en el actual.
- :w >> archivo		// Guardar el buffer al final del archivo.
- :q			// Quitar vi y volver a la terminal.
- :x			// Guardar (si ha sido modificado el archivo) y salir
- :comando!		// Evita el aviso que pueda salir al ejecutar el comando.
- :read [a] / r	[a]     // Leer el contenido del archivo a y añadirlo a partir de la línea actual.
- :ya [n]		// Copiar la línea actual en el buffer con letra n (si n es un dígito copia n líneas desde la actual).
- :pu [n]		// Pegar después de la línea actual el contenido del buffer n (puede ser dígito o letra).

- G		// Mover el cursor al final del archivo.

- Ctrl-f	// Mover la pantalla una página hacia abajo.
- Ctrl-b	// Mover la pantalla una página hacia arriba.

- [n]H		// Mover el cursor a [n líneas desde] la parte superior de la página actual.
- [n]L		// Mover el cursor a [n líneas desde] la parte inferior de la página actual.
- M		// Mover el cursor a la mitad de la página actual.
- y[m]		// Copiar lo indicado en el comando posterior de movimiento.
- [n]p		// Pegar el buffer sin nombre [n veces] después del cursor.
- [n]P		// Pegar el buffer sin nombre [n veces] antes del cursor.
- "[n]y		// Copiar en el buffer nombrado n. La n puede ser una letra o un número de un dígito. 
		// Válido también para el comando borrar d, copiando  previamente lo borrado.
		// Si la letra n es mayúscula se añade lo copiado al contenido del buffer nombrado con la letra minúscula.
- "[n]p		// Pegar lo almacenado en el buffer n. La n puede ser una letra o un número de un dígito.

- [n]I		// Insertar [n veces] texto al inicio de la línea actual.

- [n]s		// Sustituir [n caracteres desde] el carácter actual por lo nuevo a escribir en su lugar.
- o		// Añadir una línea en blanco debajo de la línea actual y entrar en modo insertar.
- O		// Añadir una línea en blanco encima de la línea actual y entrar en modo insertar.

EJECUCIÓN DE VI DESDE TERMINAL
- vi +n	archivo		// Abrir el archivo en la línea n. Se puede usar -c en lugar de +.
- vi + archivo		// Abrir el archivo en la última línea.
- vi +/patrón		// Abrir el archivo en la primera ocurrencia del patrón. Se puede usar -c en lugar de +.
- vi -R \ view archivo 	// Abrir el archivo en modo lectura.
- vi -r			// Listar todos los archivos temporales de autoguardado.
- vi -r archivo		// Recuperar el archivo de autoguardado.
